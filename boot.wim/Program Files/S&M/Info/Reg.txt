                          Настройки программы.

Находятся в реестре:
[HKEY_LOCAL_MACHINE\SOFTWARE\Serj\S&M]
"Language"                            - выбор языка. 0 = английский, 1 = русский

"RunningCounter"                      - просто счетчик запуска, не используется. Не надо вздрагивать, trial period не кончится.

"CPU0tests","CPU1tests"...            - разрешенные тесты для этого процессора

"CPUtestingTime"
Stop	  = 0
quick     = 1
norm	  = 2
slow	  = 3	
CycleMode = 4

"CPUtestingMode"
; старшие 7 бит - загрузка процессора!
MemoryHeat	= 1 ; прогрев памяти в тесте процессора
NoSafeMode	= 2 ; запрещен безопасный режим
LowQuality	= 4 ; пониженное качество тестирования
EnableLog	= 8 ; есть лог
EnableFileLog	= 16 ; запись лога в файл	
NoSplash	= 32 ;  вЫключить мерцающую рамку
CPU_SoftSafeMode    = 64  ; безопасный режим подсчета ошибок
EnableNegTemp	    = 128 ;  разрешить вывод отрицательны температур

"MemoryTestingTime"
Stop	  = 0
quick     = 1
norm	  = 2
slow	  = 3	
CycleMode = 4

"MemoryTestingMode"
SingleThread	= 1 ; тестировать 1 тредом
NoSafeMode	= 2 ; запрещен безопасный режим
VariablePattern	= 4 ; пониженное качество тестирования
RAM_SoftSafeMode= 8 ; безопасный режим подсчета ошибок

"MainWinPosX", "MainWinPosY"          - позиция основного окна

"MainWinMinimize"                     - признак, что основное окно свернуто

"MeasWinPosX", "MeasWinPosY"          - позиция окна измерений

"MeasWndActive"                       - создавать ли окно измерений

"MeasKey0"                            - кнопки включения графиков в окне измерений

"MeasKey1"                            - ...

"MeasKey2"                            - ...

"_t0", "_Xcenter","_Mx","_Xi","_My0","_My1","_Y0ofs","_Y1ofs" - позиции и множители графиков окна измерений

"_XlineStart", "_XlineEnd"            - две вертикальные линии окна измерений - начало и конец                        

"InvertX","InvertY"                   - 0=нет инверсии, соответствующие настройки окна измерений  

"Meas_Wide"                           - 0=нет, толстые линии

"CurrentTAB"                          - текущее окно/закладка основного окна

"Post80"                              - настройки, что выводить на карту Post80. Мл. 3 бита выбирают температуру 0,1,2 

"CaseDefault"                         - обычная температура в корпусе. Старший бит - переключатель Радиатор/Ядро для вывода температуры C2D

"FSAAmode"                            - соответственно. См. позиции ComboBox
; 0 = auto, 1 = NoFSAA, 2...=mode

;"BlurCount"     ; настраиваемый параметр - кол-во повторений при размазывании звездочки OpenGL
; 0 = AUTO, 1... - без автонастройки 

"LimitFPS"                           - кол-во FPS для автоматической настройки FPS

"VGAmode"
; режим настройки FPS VGA: ОСНОВНОЙ РЕЖИМ ВЫБОРА РЕЖИМА!
; 0 = AUTO, 1 = const Blur, 2 = const FPS 

"FirstRunVGA_FSAA"                    - не0 - пытаться сразу устанавливать выбранный режим FSAA, иначе - делать при повторном запуске.

"OldFSAApixelFormat"                  - копия режима пикселов для предыдущего запуска VGA 

"OldFSAA"                             - ...... и соответствующий ему номер FSAA

"ImageType"                           -  тип картинки VGA: auto/star/cube ; Tunnel=3

"TunnelType"                          - тип OGL 'Plasma': 0 = fast с большой текстурой; 1 = fast с маленькой текс.; 2 = 'Plasma Tunnel'

"FPUtestType"                         - тип теста Conroe: 0 = auto(?), 1 = AMP, 2 = Intel P4, 3 = old FPU, 4 = Intel Merom (TAT), 5 = MyConroe 

"AWEwarningCount"                     - сделал и забыл. Не используется и надеюсь не потребуется

"TransparentDelay"                    - время изменения прозрачности, mS

"GraphicsHQ"                          - 1 = повышенное качество прорисовки графиков

"UseConvMonData32_16"                 - не0 - сжимать данные мониторинга. При этом ухудшится вывод частоты и потеряется загрузка диска и памяти. Зато в 2 раза меньше. Для экономии места, программа пытается включить компрессию на файлы мониторинга и это сжатие 32->16 не столь эффективно.

"MeasGraphicMode"                     - 0 = нормал., 1 = все 3 графика окна измерений равной высоты

"DebugFlags"                          - забыл описать

"EnableRAMtests"                      - тесты памяти, доступные для работы. Всего их 4, не надо ставить больше.

"HDDtestMode"                         - мл. байт номер позиции ComboBox выбора диска, байт+1 - биты включения тестов HDD

"CaseCompensation"                    - компенсация температуры процессора C2* для перевода температуры из ядра в радиатор.

"RedUsageType"                        - что хранить в 'красном' графике мониторинга.
;RedUsageType                    ; тип красного графика загрузки - диск или пиковая загрузка процессора
;                                ; d3..0 = время с'ема данных(сек), 0=0.5сек
;                                ; d7..4 = тип преобразования числа. HDD: 0=linear, 1=log 50db, 2=100db
;                                ; d11..d8 = особенность выбора данных. HDD: 0=RW, 1=R, 2=W, 3=none; PeakUsage: 0 = максимум сумма_CPU(i)/CPU, 1 = максимум из процессоров 
;                                ; d15..12 = источник. 0=disable, 1=Peak CPU usage, 2=HDD
